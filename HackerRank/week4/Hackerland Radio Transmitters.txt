# include <stdio.h>
#include <stdlib.h>
int compare(const void *a,const void *b){
    return(*(int*)a-*(int*)b);
}
int min_1(int q_1[],int total_num,int k){
    qsort(q_1,total_num,sizeof(int),compare);
    int add_1 =0;
    int i =0;
    while(i<total_num){
        add_1++;
        int index_1 =q_1[i]+k;
        while(i<total_num && q_1[i]<= index_1) i++;
        index_1 = q_1[i-1]+k;
        while(i<total_num&&q_1[i]<= index_1) i++;
    }
    return add_1;
}
int main(){
    int q_2,e;
    scanf("%d %d",&q_2,&e);
    int home[q_2];
    for(int i=0;i<q_2;i++){
        scanf("%d",&home[i]);
    }
    int last = min_1(home,q_2,e);
    printf("%d\n",last);
    return 0;
}
