#include<stdio.h>
int main(){
   int n,i,c;
   scanf("%d",&n);
   char str[n+1];
   scanf("%s",str);
   int m;
   scanf("%d",&m);
   m=m%26;
   for(i=0;i<n;i++){
     c=str[i];
     
     if(c>='A' && c<='Z'){
        c=c+m;
        if (c>'Z'){
            c=c-26;
        }
        
     }
     else if(c>='a' && c<='z'){
        c=c+m;
        if(c>'z'){
            c=c-26;
        }
        
     }
     str[i]=c;
   }
   printf("%s",str);

}
