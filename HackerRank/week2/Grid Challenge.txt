#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    int n,i,j,k,dumpy,temp=1;
    scanf("%d",&n);
    char arr[100][100];
    for(i=0;i<n;i++){
        scanf("%s",arr[i]);
        
    }
    for(i=0;i<n;i++){
        for(k=0;k<n-1;k++){
        for(j=0;j<n-k-1;j++){
           
            if(arr[i][j]>arr[i][j+1]){
                 char dumpy= arr[i][j];
                 arr[i][j]=arr[i][j+1];
                 arr[i][j+1]=dumpy;
                
            }
            
        }
        }
    }
    
    for(j=0;j<n;j++){
        for(i=0;i<n-1;i++){
            if(arr[i][j]>arr[i+1][j]){
                temp=0;
                break;
                
                
            }
        }
        if (temp==0){
            break;
        }
    }
    if(temp){
        printf("YES\n");
    }
    else{
        printf("NO\n");
    }
    
    
}

}
