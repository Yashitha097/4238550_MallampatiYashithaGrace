#include<stdio.h>
int main(){
    int n,i,j,count,divid,mark=0;
    scanf("%d",&n);
    int arr[n],visited[n];
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
        visited[i]=0;
        
    }
    for(i=0;i<n;i++){
        count=0;
        if (visited[i]==1){
            continue;
        }
       for(j=0;j<n;j++){
        if(arr[i]==arr[j]){
            count=count+1;
            visited[j]=1;
        }
        
       }
       divid=count/2;
       mark=mark+divid;
       
      
    }
    printf("%d",mark);
}
