#include<stdio.h>
#include<stdlib.h>
int cmpfunc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}
int main(){
    int n,m,i,j,start,end,difference_value,min_value=1000000000,temp;
    scanf("%d %d",&n,&m);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        
    }
     qsort(a, n, sizeof(int), cmpfunc);
    for(i=0;(i<=n-m);i++){
        start=a[i];
        end=a[i+m-1];
        difference_value=end-start;
        if (difference_value<min_value){
            min_value=difference_value;
        }
        
        
    }
    printf("%d",min_value);
}
