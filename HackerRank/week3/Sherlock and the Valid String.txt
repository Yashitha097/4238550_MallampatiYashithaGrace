#include <stdio.h>
#include <string.h>
int main(){
    char string_1[100005];
    int c[26] = {0};
    int i,length_1,count_1[26],number_1=0,m=0,c_f=0,c_c=0;
    scanf("%s",string_1);
    length_1=strlen(string_1);
    for(i=0;i<length_1;i++){
        c[string_1[i]-'a']++;
        
    }
    
    for(i=0;i<26;i++){
        if(c[i]>0){
            count_1[number_1++]=c[i];
        }
    }
    
    for(i=0;i<number_1;i++){
        int count=0;
        for(int j=0;j<number_1;j++){
            if(count_1[i]==count_1[j]){
                count++;
            }
        }
        if(count > c_c){
            c_c = count;
            c_f=count_1[i];
        }
    }
    
    for(i=0;i<number_1;i++){
        if(count_1[i] != c_f){
            if(count_1[i] != c_f){
                if(count_1[i]==1){
                    m++;
                }
                else if (count_1[i]==c_f+1){
                    m++;
                }
                else{
                    m=2;
                    break;
                }
            }
        }
        
    }
    if (m <= 1){
            printf("YES\n");
        }
    else{
            printf("NO\n");
        }
    return 0;
}
