char* isValid(char* s) {
    static char YES[] = "YES";
    static char NO[] = "NO";
    int repeat[26],i,store,repeat_main[26],max_value,min_value,min_C =0,max_C =0;;
    
    for(i=0;i<26;i++){
        repeat[i]=0;
    }
    for(i=0;s[i] != '\0';i++){
        store =s[i]- 'a';
        repeat[store] = repeat[store] +1;
    }
    
    int count =0;
    for(i=0;i<26;i++){
        if(repeat[i]>0){
            repeat_main[count] = repeat[i];
            count++;
        }
    }
     min_value = repeat_main[0];
     max_value =repeat_main[0];
    for(i=1;i<count;i++){
        if(repeat_main[i]<min_value){
            min_value=repeat_main[i];
        }
        if(repeat_main[i]>max_value){
            max_value=repeat_main[i];
        }
    }
    if (min_value==max_value){
        return YES;
    }
   
    for(i=0;i<count;i++){
        if(repeat_main[i]==min_value){
            min_C++;
        }
        if (repeat_main[i]==max_value){
            max_C++;
        }
    }
    if(min_value==1 && min_C==1 && max_C == count -1){
        return YES;
    }
    if ((max_value-min_value==1) && max_C ==1){
        return YES;
    }
    return NO;

}

char* readline() {
    size_t alloc_length = 1024;
    size_t data_length = 0;

    char* data = malloc(alloc_length);

    while (true) {
        char* cursor = data + data_length;
        char* line = fgets(cursor, alloc_length - data_length, stdin);

        if (!line) {
            break;
        }

        data_length += strlen(cursor);

        if (data_length < alloc_length - 1 || data[data_length - 1] == '\n') {
            break;
        }

        alloc_length <<= 1;

        data = realloc(data, alloc_length);

        if (!data) {
            data = '\0';

            break;
        }
    }

    if (data[data_length - 1] == '\n') {
        data[data_length - 1] = '\0';

        data = realloc(data, data_length);

        if (!data) {
            data = '\0';
        }
    } else {
        data = realloc(data, data_length + 1);

        if (!data) {
            data = '\0';
        } else {
            data[data_length] = '\0';
        }
    }

    return data;
}
