#include<stdio.h>
int main(){
    int n,i,j,sum_num,count_num=0;
    scanf("%d",&n);
    int arr[n],d,m;
    for (i=0;i<(n);i++){
        scanf("%d",&arr[i]);
    }
    scanf("%d %d",&d,&m);
    for (i=0;i<=(n-m);i++){
        sum_num=0;
        for (j=0;j<m;j++){
            sum_num=sum_num+arr[i+j];
            
        }
        if (sum_num==d){
            count_num=count_num+1;
            
        }
        
    }
    printf("%d ",count_num);
   
   
}
